} "Email" et "Mot de passe" en 	font: noir inherit; size 12 gras, un bouton "Se connecter" en 	font-family: 'Syne';
	font-weight: 700;
	font-size: 30px;
	background color: #1D6154
	text color : white
un texte "Mot de passe oublié" souligné en noir inherit size 12 gras le tout centre au milieu de la modale 
Lorsque le couple identifiant et mot de passe n’est pas bon pour se connecter il faut afficher le message d’erreur: 

“Erreur dans l’identifiant ou le mot de passe”

Lorsque le couple identifiant et mot de passe est correct, alors il faut rediriger vers la page du site avec cette fois ci des boutons d’actions pour éditer le site.

document.addEventListener("DOMContentLoaded", () => {
    const loginLink = document.querySelector("nav ul li:nth-child(3)"); // login
    if (!loginLink) return;

    loginLink.addEventListener("click", (e) => {
        e.preventDefault();
        console.log("Log in"); // ✅ console log

        // Crée la modale
        createLoginModal();
    });

    function createLoginModal() {
        // Overlay
        const overlay = document.createElement("div");
        overlay.style.position = "fixed";
        overlay.style.top = 0;
        overlay.style.left = 0;
        overlay.style.width = "100%";
        overlay.style.height = "100%";
        overlay.style.background = "rgba(0,0,0,0.5)";
        overlay.style.display = "flex";
        overlay.style.justifyContent = "center";
        overlay.style.alignItems = "center";
        overlay.style.zIndex = "999";

        // Conteneur modale
        const modal = document.createElement("div");
        modal.style.background = "#fff";
        modal.style.padding = "30px";
        modal.style.borderRadius = "10px";
        modal.style.textAlign = "center";
        modal.style.width = "400px";

        // Titre
        const title = document.createElement("h2");
        title.innerText = "Log in";
        title.style.fontFamily = "'Syne', sans-serif";
        title.style.fontWeight = "700";
        title.style.fontSize = "30px";
        title.style.color = "#1D6154";

        // Formulaire
        const form = document.createElement("form");

        // Label email
        const labelEmail = document.createElement("label");
        labelEmail.innerText = "Email";
        labelEmail.style.display = "block";
        labelEmail.style.fontSize = "12px";
        labelEmail.style.fontWeight = "bold";
        labelEmail.style.color = "black";
        labelEmail.style.marginTop = "10px";

        const inputEmail = document.createElement("input");
        inputEmail.type = "email";
        inputEmail.style.width = "100%";
        inputEmail.style.padding = "10px";
        inputEmail.style.marginTop = "5px";
        inputEmail.style.border = "1px solid #ccc";

        // Label mot de passe
        const labelPassword = document.createElement("label");
        labelPassword.innerText = "Mot de passe";
        labelPassword.style.display = "block";
        labelPassword.style.fontSize = "12px";
        labelPassword.style.fontWeight = "bold";
        labelPassword.style.color = "black";
        labelPassword.style.marginTop = "15px";

        const inputPassword = document.createElement("input");
        inputPassword.type = "password";
        inputPassword.style.width = "100%";
        inputPassword.style.padding = "10px";
        inputPassword.style.marginTop = "5px";
        inputPassword.style.border = "1px solid #ccc";

        // Message d'erreur
        const errorMsg = document.createElement("p");
        errorMsg.style.color = "red";
        errorMsg.style.fontWeight = "bold";
        errorMsg.style.marginTop = "10px";
        errorMsg.style.display = "none";

        // Bouton "Se connecter"
        const btnLogin = document.createElement("button");
        btnLogin.type = "submit";
        btnLogin.innerText = "Se connecter";
        btnLogin.style.fontFamily = "'Syne', sans-serif";
        btnLogin.style.fontWeight = "700";
        btnLogin.style.fontSize = "30px";
        btnLogin.style.backgroundColor = "#1D6154";
        btnLogin.style.color = "white";
        btnLogin.style.border = "none";
        btnLogin.style.padding = "10px 20px";
        btnLogin.style.marginTop = "20px";
        btnLogin.style.cursor = "pointer";

        // "Mot de passe oublié"
        const forgotPwd = document.createElement("p");
        forgotPwd.innerText = "Mot de passe oublié";
        forgotPwd.style.textDecoration = "underline";
        forgotPwd.style.color = "black";
        forgotPwd.style.fontSize = "12px";
        forgotPwd.style.fontWeight = "bold";
        forgotPwd.style.marginTop = "15px";
        forgotPwd.style.cursor = "pointer";

        // Ajout au formulaire
        form.appendChild(labelEmail);
        form.appendChild(inputEmail);
        form.appendChild(labelPassword);
        form.appendChild(inputPassword);
        form.appendChild(errorMsg);
        form.appendChild(btnLogin);
        form.appendChild(forgotPwd);

        // Fermeture en cliquant hors de la modale
        overlay.addEventListener("click", (e) => {
            if (e.target === overlay) overlay.remove();
        });

        // Gestion du submit
        form.addEventListener("submit", async (e) => {
            e.preventDefault();
            const email = inputEmail.value.trim();
            const password = inputPassword.value.trim();

            try {
                const response = await fetch("http://localhost:5678/api/users/login", {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({ email, password })
                });

                if (!response.ok) {
                    errorMsg.innerText = "Erreur dans l’identifiant ou le mot de passe";
                    errorMsg.style.display = "block";
                    return;
                }

                const data = await response.json();
                localStorage.setItem("token", data.token);

                // Redirection
                window.location.href = "edition.html"; // page d’édition
            } catch (error) {
                console.error("Erreur connexion :", error);
                errorMsg.innerText = "Erreur serveur";
                errorMsg.style.display = "block";
            }
        });

        // Ajout dans la page
        modal.appendChild(title);
        modal.appendChild(form);
        overlay.appendChild(modal);
        document.body.appendChild(overlay);
    }
});
